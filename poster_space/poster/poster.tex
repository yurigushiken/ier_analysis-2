% =====================================================================
\begin{column}{\colwidth}

  \begin{block}{Introduction}

    How do pre-linguistic infants parse events into the argument structures required for language?

    \begin{itemize}
      \item \textbf{Home signers} evidence: deaf children of hearing parents invent gestural communication systems with argument structure, even without linguistic input (Goldin-Meadow, 1985).
      \item This suggests the human conceptual system constructs event representations with argument-like structures \textit{independent of language}.
      \item \textbf{Trivalent events} like \textit{giving} require three arguments [source, theme, goal], while \textit{hugging} requires only two [agent, patient]. Do infants distinguish these?
    \end{itemize}

    \begin{figure}
      \centering
      \includegraphics[width=0.50\colwidth]{figures/gordon2003_fig1.png}
      \caption{Mapping relations between events and language in the acquisition of verb-argument structure (Gordon, 2003).}
    \end{figure}

  \end{block}

  \begin{alertblock}{Prior Work: Gordon (2003)}

    Using a \textbf{habituation paradigm} with looped videos of GIVE and HUG events:

    \begin{itemize}
      \item \textbf{10-month-olds} dishabituated when the toy was removed from GIVE but \textit{not} from HUG, indicating the toy is relevant to GIVE's argument structure.
      \item \textbf{8-month-olds} showed a marginal effect ($p$=.07); \textbf{6-month-olds} showed no effect.
      \item \textbf{SHOW} (3 arguments, but more complex): even 10-month-olds failed to dishabituate, suggesting information transfer requires more advanced conceptualization.
    \end{itemize}

    \begin{figure}
      \centering
      \includegraphics[width=0.48\colwidth]{figures/gordon2003_timelapse.png}
      \caption{Stimulus video frames from Gordon (2003). Rows show GIVE with toy, GIVE without toy, HUG with toy, and HUG without toy conditions.}
    \end{figure}

    \begin{figure}
      \centering
      \includegraphics[width=0.52\colwidth]{figures/gordon2003_age_changes.png}
      \caption{Age-related changes in looking time on test trials for GIVE and HUG across 6, 8, and 10 months (Gordon, 2003).}
    \end{figure}

  \end{alertblock}

  \begin{block}{Research Questions}

    \begin{enumerate}
      \item How do gaze scanning strategies shift across 7 to 11 months of age?
      \item Do infants transition from tracking object movement (Toy$\leftrightarrow$Body) to attending to faces (Face$\leftrightarrow$Face)?
      \item Do GIVE and SHOW conditions differ in developmental trajectory toward adult-like patterns?
    \end{enumerate}

  \end{block}

\end{column}

\begin{column}{\colwidth}

  \begin{block}{Methods}

    \heading{Participants}

    40 infants (7 to 11 months) and 15 adults in a within-subjects design. Participant counts vary by condition due to data quality (GIVE: $n$=40; SHOW: $n$=45).

    \heading{Stimuli \& EOI Coding}

    Looped videos of GIVE and SHOW events. Rather than traditional screen-based AOIs, we used a custom computer-vision pipeline (\textbf{Eye-Track-ML}; Gushiken, Li, \& Gordon, 2025) to automatically detect and segment event elements frame-by-frame as objects, which we call \textbf{Elements of Interest (EOIs)}.

    \begin{center}
    \begin{minipage}[c]{0.46\colwidth}
      \begin{figure}
        \centering
        \includegraphics[width=\textwidth]{figures/eoi_stimulus.png}
        \caption{GIVE stimulus frame with Eye-Track-ML SAM overlay showing segmented EOIs: man, woman, and toy.}
      \end{figure}
    \end{minipage}%
    \hspace{0.04\colwidth}%
    \begin{minipage}[c]{0.15\colwidth}
      \centering
      \includegraphics[width=0.75\textwidth]{figures/eyetrackml_qr.png}\\[0.5ex]
      {\scriptsize \textbf{Eye-Track-ML}\\Gushiken et al.,\\2025}
    \end{minipage}
    \end{center}

    \heading{Statistical Approach}

    Precision-weighted Gaussian GEE with linear age trends across infant cohorts; adults as reference group. Fixation threshold: $\geq$3 consecutive frames; trial inclusion: $\geq$50\% on-screen looking.

  \end{block}

  \begin{alertblock}{GIVE vs.\ SHOW Comparison}

    \begin{table}
      \centering
      \begin{tabular}{l c c c c}
        \toprule
        \textbf{Transition} & \textbf{Cond.} & $\boldsymbol{\beta}$ & \textbf{$p$} & \textbf{11-mo vs.\ Adult} \\
        \midrule
        Face$\leftrightarrow$Face & GIVE & 0.053 & .001 & 20.3\% vs.\ 24.4\% \\
        Face$\leftrightarrow$Face & SHOW & 0.033 & .007 & 15.8\% vs.\ 34.3\% \\
        \midrule
        Toy$\leftrightarrow$Body & GIVE & $-$0.078 & .006 & \\
        Toy$\leftrightarrow$Body & SHOW & $-$0.075 & \textless .0001 & \\
        \midrule
        Toy$\leftrightarrow$Face & GIVE & 0.045 & .088 & marginal \\
        Toy$\leftrightarrow$Face & SHOW & 0.057 & \textless .0001 & significant \\
        \bottomrule
      \end{tabular}
      \caption{Summary of linear age trends across GIVE and SHOW conditions.}
    \end{table}

    Infants approach adult-like gaze patterns for GIVE by 11 months but \textit{not} for SHOW, paralleling Gordon (2003) habituation findings.

  \end{alertblock}

  \begin{block}{Take-Away}

    \textbf{Infants develop adult-like event parsing for GIVE by 11 months but not for SHOW}, suggesting that \textit{transfer of possession} is understood before \textit{information transfer}. This converges with Gordon (2003), where infants dishabituated to toy removal in GIVE but not SHOW. The shift from Toy$\leftrightarrow$Body to Face$\leftrightarrow$Face transitions may reflect emerging understanding of \textbf{intentionality} and may scaffold verb-argument structure acquisition.

  \end{block}
